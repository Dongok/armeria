41  /**
42   * A parser of the raw path and query components of an HTTP path. Performs validation and allows caching of
43   * results.
44   */
45  public final class PathAndQuery {
46  
47      private static final PathAndQuery ROOT_PATH_QUERY = new PathAndQuery("/", null);
48  
49      private static final BitSet ALLOWED_PATH_CHARS = new BitSet();
50      private static final BitSet ALLOWED_QUERY_CHARS = new BitSet();
51  
52      static {
53          final String allowedPathChars =
54                  "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789-._~:/?#[]@!$&'()*+,;=";
55          for (int i = 0; i < allowedPathChars.length(); i++) {
56              ALLOWED_PATH_CHARS.set(allowedPathChars.charAt(i));
57          }
58  
59          final String allowedQueryChars =
60                  "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789-._~:/?#[]@!$&'()*,;=";
61          for (int i = 0; i < allowedQueryChars.length(); i++) {
62              ALLOWED_QUERY_CHARS.set(allowedQueryChars.charAt(i));
63          }
64      }

101     /**
102      * Validates the {@link String} that contains an absolute path and a query, and splits them into
103      * the path part and the query part. If the path is usable (e.g., can be served a successful response from
104      * the server and doesn't have variable path parameters), {@link PathAndQuery#storeInCache(String)} should
105      * be called to cache the parsing result for faster future invocations.
106      *
107      * @return a {@link PathAndQuery} with the absolute path and query, or {@code null} if the specified
108      *         {@link String} is not an absolute path or invalid.
109      */
110     @Nullable
111     public static PathAndQuery parse(@Nullable String rawPath) {
112         if (CACHE != null && rawPath != null) {
113             final PathAndQuery parsed = CACHE.getIfPresent(rawPath);
114             if (parsed != null) {
115                 return parsed;
116             }
117         }
118         return splitPathAndQuery(rawPath);
119     }
120 
121     /**
122      * Stores this {@link PathAndQuery} into cache for the given raw path. This should be used by callers when
123      * the parsed result was valid (e.g., when a server is able to successfully handle the parsed path).
124      */
125     public void storeInCache(@Nullable String rawPath) {
126         if (CACHE != null && rawPath != null) {
127             CACHE.put(rawPath, this);
128         }
129     }

177     @Nullable
178     private static PathAndQuery splitPathAndQuery(@Nullable final String pathAndQuery) {
179         final Bytes path;
180         final Bytes query;
181 
182         if (pathAndQuery == null) {
183             return ROOT_PATH_QUERY;
184         }
185 
186         // Split by the first '?'.
187         final int queryPos = pathAndQuery.indexOf('?');
188         if (queryPos >= 0) {
189             if ((path = decodePercentsAndEncodeToUtf8(
190                     pathAndQuery, 0, queryPos, true)) == null) {
191                 return null;
192             }
193             if ((query = decodePercentsAndEncodeToUtf8(
194                     pathAndQuery, queryPos + 1, pathAndQuery.length(), false)) == null) {
195                 return null;
196             }
197         } else {
198             if ((path = decodePercentsAndEncodeToUtf8(
199                     pathAndQuery, 0, pathAndQuery.length(), true)) == null) {
200                 return null;
201             }
202             query = null;
203         }
204 
205         if (path.data[0] != '/') {
206             // Do not accept a relative path.
207             return null;
208         }
209 
210         // Reject the prohibited patterns.
211         if (firstPathComponentContainsColon(path) || pathContainsDoubleDots(path)) {
212             return null;
213         }
214 
215         return new PathAndQuery(encodeToPercents(path, true),
216                                 query != null ? encodeToPercents(query, false) : null);
217     }

219     @Nullable
220     private static Bytes decodePercentsAndEncodeToUtf8(String value, int start, int end,
221                                                           boolean isPath) {
222         final int length = end - start;
223         if (length == 0) {
224             return isPath ? ROOT_PATH : EMPTY_QUERY;
225         }
226 
227         final Bytes buf = new Bytes(Math.max(length * 3 / 2, 4));
228         int lastCP = 0;
229         for (final CodePointIterator i = new CodePointIterator(value, start, end); i.hasNextCodePoint();) {
230             final int pos = i.position();
231             final int cp = i.nextCodePoint();
232 
233             if (cp == '%') {
234                 final int hexEnd = pos + 3;
235                 if (hexEnd > end) {
236                     // '%' or '%x' (must be followed by two hexadigits)
237                     return null;
238                 }
239 
240                 final char digit1 = value.charAt(pos + 1);
241                 final char digit2 = value.charAt(pos + 2);
242                 if (!isHexadigit(digit1) || !isHexadigit(digit2)) {
243                     // The first or second digit is not hexadecimal.
244                     return null;
245                 }
246 
247                 final int decoded = (decodeHexNibble(digit1) << 4) | decodeHexNibble(digit2);
248                 if (!appendOneByte(buf, decoded, lastCP, isPath)) {
249                     return null;
250                 }
251                 i.position(hexEnd);
252                 lastCP = decoded;
253                 continue;
254             }
255 
256             if (cp == '+' && !isPath) {
257                 buf.ensure(1);
258                 buf.add((byte) ' ');
259                 lastCP = '+';
260                 continue;
261             }
262 
263             if (cp <= 0x7F) {
264                 if (!appendOneByte(buf, cp, lastCP, isPath)) {
265                     return null;
266                 }
267             } else if (cp <= 0x7ff) {
268                 buf.ensure(2);
269                 buf.add((byte) ((cp >>> 6) | 0b110_00000));
270                 buf.add((byte) (cp & 0b111111 | 0b10_000000));
271             } else if (cp <= 0xffff) {
272                 buf.ensure(3);
273                 buf.add((byte) ((cp >>> 12) | 0b1110_0000));
274                 buf.add((byte) (((cp >>> 6) & 0b111111) | 0b10_000000));
275                 buf.add((byte) ((cp & 0b111111) | 0b10_000000));
276             } else if (cp <= 0x1fffff) {
277                 buf.ensure(4);
278                 buf.add((byte) ((cp >>> 18) | 0b11110_000));
279                 buf.add((byte) (((cp >>> 12) & 0b111111) | 0b10_000000));
280                 buf.add((byte) (((cp >>> 6) & 0b111111) | 0b10_000000));
281                 buf.add((byte) ((cp & 0b111111) | 0b10_000000));
282             } else if (cp <= 0x3ffffff) {
283                 // A valid unicode character will never reach here, but for completeness.
284                 // http://unicode.org/mail-arch/unicode-ml/Archives-Old/UML018/0330.html
285                 buf.ensure(5);
286                 buf.add((byte) ((cp >>> 24) | 0b111110_00));
287                 buf.add((byte) (((cp >>> 18) & 0b111111) | 0b10_000000));
288                 buf.add((byte) (((cp >>> 12) & 0b111111) | 0b10_000000));
289                 buf.add((byte) (((cp >>> 6) & 0b111111) | 0b10_000000));
290                 buf.add((byte) ((cp & 0b111111) | 0b10_000000));
291             } else {
292                 // A valid unicode character will never reach here, but for completeness.
293                 // http://unicode.org/mail-arch/unicode-ml/Archives-Old/UML018/0330.html
294                 buf.ensure(6);
295                 buf.add((byte) ((cp >>> 30) | 0b1111110_0));
296                 buf.add((byte) (((cp >>> 24) & 0b111111) | 0b10_000000));
297                 buf.add((byte) (((cp >>> 18) & 0b111111) | 0b10_000000));
298                 buf.add((byte) (((cp >>> 12) & 0b111111) | 0b10_000000));
299                 buf.add((byte) (((cp >>> 6) & 0b111111) | 0b10_000000));
300                 buf.add((byte) ((cp & 0b111111) | 0b10_000000));
301             }
302 
303             lastCP = cp;
304         }
305 
306         return buf;
307     }

309     private static boolean appendOneByte(Bytes buf, int cp, int lastCP, boolean isPath) {
310         if (cp == 0x7F || (cp >>> 5 == 0)) {
311             // Reject the prohibited control characters: 0x00..0x1F and 0x7F
312             return false;
313         }
314 
315         if (cp == '/' && isPath) {
316             if (lastCP != '/') {
317                 buf.ensure(1);
318                 buf.add((byte) '/');
319             } else {
320                 // Remove the consecutive slashes: '/path//with///consecutive////slashes'.
321             }
322         } else {
323             buf.ensure(1);
324             buf.add((byte) cp);
325         }
326 
327         return true;
328     }

330     private static boolean isHexadigit(char ch) {
331         return ch >= '0' && ch <= '9' ||
332                ch >= 'a' && ch <= 'f' ||
333                ch >= 'A' && ch <= 'F';
334     }

336     /**
337      * According to RFC 3986 section 3.3, path can contain a colon, except the first segment.
338      *
339      * <p>Should allow the asterisk character in the path, query, or fragment components of a URL(RFC2396).
340      * @see <a href="https://tools.ietf.org/html/rfc3986#section-3.3">RFC 3986, section 3.3</a>
341      */
342     private static boolean firstPathComponentContainsColon(Bytes path) {
343         final int length = path.length;
344         for (int i = 1; i < length; i++) {
345             final byte b = path.data[i];
346             if (b == '/') {
347                 break;
348             }
349             if (b == ':') {
350                 return true;
351             }
352         }
353         return false;
354     }
355 
356     private static boolean pathContainsDoubleDots(Bytes path) {
357         final int length = path.length;
358         byte b0 = 0;
359         byte b1 = 0;
360         byte b2 = '/';
361         for (int i = 1; i < length; i++) {
362             final byte b3 = path.data[i];
363             if (b3 == '/' && b2 == '.' && b1 == '.' && b0 == '/') {
364                 return true;
365             }
366             b0 = b1;
367             b1 = b2;
368             b2 = b3;
369         }
370 
371         return b0 == '/' && b1 == '.' && b2 == '.';
372     }

374     private static String encodeToPercents(Bytes value, boolean isPath) {
375         final BitSet allowedChars = isPath ? ALLOWED_PATH_CHARS : ALLOWED_QUERY_CHARS;
376         final int length = value.length;
377         boolean needsEncoding = false;
378         for (int i = 0; i < length; i++) {
379             if (!allowedChars.get(value.data[i] & 0xFF)) {
380                 needsEncoding = true;
381                 break;
382             }
383         }
384 
385         if (!needsEncoding) {
386             // Deprecated, but it fits perfect for our use case.
387             //noinspection deprecation
388             return new String(value.data, 0, 0, length);
389         }
390 
391         final StringBuilder buf = new StringBuilder(length);
392         for (int i = 0; i < length; i++) {
393             final int b = value.data[i] & 0xFF;
394             if (allowedChars.get(b)) {
395                 buf.append((char) b);
396             } else if (b == '+' && !isPath) {
397                 buf.append("%2B");
398             } else if (b == ' ') {
399                 if (isPath) {
400                     buf.append("%20");
401                 } else {
402                     buf.append('+');
403                 }
404             } else {
405                 buf.append('%');
406                 appendHexNibble(buf, b >>> 4);
407                 appendHexNibble(buf, b & 0xF);
408             }
409         }
410 
411         return buf.toString();
412     }
413 
414     private static void appendHexNibble(StringBuilder buf, int nibble) {
415         if (nibble < 10) {
416             buf.append((char) ('0' + nibble));
417         } else {
418             buf.append((char) ('A' + nibble - 10));
419         }
420     }

422     private static final class Bytes {
423         byte[] data;
424         int length;
425 
426         Bytes(int initialCapacity) {
427             data = new byte[initialCapacity];
428         }
429 
430         Bytes(byte[] data) {
431             this.data = data;
432             length = data.length;
433         }
434 
435         void add(byte b) {
436             data[length++] = b;
437         }
438 
439         void ensure(int numBytes) {
440             int newCapacity = length + numBytes;
441             if (newCapacity <= data.length) {
442                 return;
443             }
444 
445             newCapacity =
446                     (int) Math.max(Math.min((long) data.length + (data.length >> 1), Arrays.MAX_ARRAY_SIZE),
447                                    newCapacity);
448 
449             data = ByteArrays.forceCapacity(data, newCapacity, length);
450         }
451     }

453     private static final class CodePointIterator {
454         private final CharSequence str;
455         private final int end;
456         private int pos;
457 
458         CodePointIterator(CharSequence str, int start, int end) {
459             this.str = str;
460             this.end = end;
461             pos = start;
462         }
463 
464         int position() {
465             return pos;
466         }
467 
468         void position(int pos) {
469             this.pos = pos;
470         }
471 
472         boolean hasNextCodePoint() {
473             return pos < end;
474         }
475 
476         int nextCodePoint() {
477             assert pos < end;
478 
479             final char c1 = str.charAt(pos++);
480             if (Character.isHighSurrogate(c1) && pos < end) {
481                 final char c2 = str.charAt(pos);
482                 if (Character.isLowSurrogate(c2)) {
483                     pos++;
484                     return Character.toCodePoint(c1, c2);
485                 }
486             }
487 
488             return c1;
489         }
490     }
